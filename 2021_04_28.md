# Creación de la función calculadora

```python
def calculadora(operacion, x, y):
  """
  Resuelve la operación aritmética correspondiente para los valores x e y
  Args:
    operation: String que indica la operación aritmética a realizar (sumar, restar, producto, division)
    x: Número real
    y: Número real
  Returns:
    result: Número real solución de la operación aritmética indicada para los valores x e y
  """
  operacion = operacion.lower()
  if operacion == "sumar":
    result = x + y
  elif operacion == "restar":
    result = x - y
  elif operacion == "producto":
    result = x * y
  elif operacion == "division":
    result = x / y
  else:
    print("Has indicado una operación que esta calculadora no puede resolver.")
    result = "NULL"

  if result != "NULL":
    helper(operacion, x, y)
    return result

def helper(operacion, x, y):
  """
  Indica la operación realizada e imprime el resultado
  Args:
    operacion: String que indica la operación aritmética a realizar (sumar, restar, producto, division)
    x: Número real
    y: Número real
  """
  if operacion == "sumar":
    sign = "+"
  elif operacion == "restar":
    sign = "-"
  elif operacion == "producto":
    sign = "x"
  elif operacion == "division":
    sign = ":"

  print("Se ha realizado la operación {} {} {}".format(x, sign, y))
  ```
